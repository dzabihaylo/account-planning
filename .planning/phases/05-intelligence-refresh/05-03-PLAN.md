```javascript
function getRefreshStalenessClass(lastRefreshedAt) {
  if (!lastRefreshedAt) return 'stale-stale';
  var now = new Date();
  var refreshed = new Date(lastRefreshedAt);
  var days = Math.floor((now - refreshed) / (1000 * 60 * 60 * 24));
  if (days < 7) return 'stale-fresh';
  if (days < 30) return 'stale-aging';
  return 'stale-stale';
}

function getRefreshStalenessLabel(lastRefreshedAt) {
  if (!lastRefreshedAt) return 'Never refreshed';
  var now = new Date();
  var refreshed = new Date(lastRefreshedAt);
  var days = Math.floor((now - refreshed) / (1000 * 60 * 60 * 24));
  if (days === 0) return 'Refreshed today';
  if (days === 1) return 'Refreshed 1 day ago';
  return 'Refreshed ' + days + ' days ago';
}
```

**4. Add manualRefresh function.** Add in the `<script>` section:

```javascript
async function manualRefresh(accountId) {
  var btn = document.getElementById('refresh-btn-' + accountId);
  var badge = document.getElementById('refresh-badge-' + accountId);
  if (!btn) return;

  // Loading state (D-18)
  btn.disabled = true;
  btn.textContent = 'Refreshing...';

  try {
    var resp = await fetch('/api/accounts/' + accountId + '/refresh', { method: 'POST' });
    var data = await resp.json();

    if (!resp.ok) {
      throw new Error(data.error || 'Refresh failed');
    }

    // Update badge (D-18)
    if (badge) {
      badge.className = 'stale-badge ' + getRefreshStalenessClass(data.account.last_refreshed_at);
      badge.textContent = getRefreshStalenessLabel(data.account.last_refreshed_at);
    }

    // Update account data in local array (D-20)
    var idx = accounts.findIndex(function(a) { return a.id === accountId; });
    if (idx !== -1) {
      accounts[idx] = data.account;
    }

    // Re-render overview tab with updated context (D-20)
    var overviewPane = document.getElementById(accountId + '-overview');
    if (overviewPane) {
      overviewPane.innerHTML = '<div class="hl"><div class="hl-label">Account Intelligence</div>' + escapeHtml(data.account.context || '').replace(/\n/g, '<br>') + '</div>';
    }

    // Update KPI values if changed (D-22)
    var panel = document.getElementById('panel-' + accountId);
    if (panel) {
      var kpiVals = panel.querySelectorAll('.kpi-val');
      if (kpiVals.length >= 2) {
        kpiVals[0].textContent = data.account.revenue || '';
        kpiVals[1].textContent = data.account.employees || '';
      }
    }

    // Show changes summary toast
    showRefreshToast(data.account.name, data.changes_summary);

    // Refresh budget indicator
    loadBudgetStatus();

  } catch (e) {
    showRefreshToast('Refresh Error', e.message || 'There was an error refreshing this account.');
  } finally {
    btn.disabled = false;
    btn.textContent = 'Refresh Intelligence';
  }
}

function showRefreshToast(title, message) {
  // Remove existing toast if any
  var existing = document.querySelector('.refresh-toast');
  if (existing) existing.remove();

  var toast = document.createElement('div');
  toast.className = 'refresh-toast';
  toast.innerHTML = '<div class="refresh-toast-title">' + escapeHtml(title) + '</div><div class="refresh-toast-body">' + escapeHtml(message) + '</div>';
  document.body.appendChild(toast);

  // Auto-dismiss after 8 seconds
  setTimeout(function() {
    if (toast.parentNode) toast.remove();
  }, 8000);

  // Click to dismiss
  toast.addEventListener('click', function() { toast.remove(); });
}
```

```javascript
async function loadBudgetStatus() {
  try {
    var resp = await fetch('/api/refresh/budget');
    if (!resp.ok) return;
    var data = await resp.json();

    var indicator = document.getElementById('budgetIndicator');
    var text = document.getElementById('budgetText');
    var fill = document.getElementById('budgetFill');
    if (!indicator || !text || !fill) return;

    // Only show when usage exceeds 50% (Claude's discretion -- avoid cluttering clean UI)
    if (data.pct < 50) {
      indicator.style.display = 'none';
      return;
    }

    indicator.style.display = 'inline-flex';
    text.textContent = data.pct + '% budget';
    fill.style.width = Math.min(data.pct, 100) + '%';
    fill.className = 'budget-fill ' + (data.pct < 70 ? 'ok' : data.pct < 90 ? 'warn' : 'danger');
  } catch (e) {
    // Silently fail -- budget indicator is informational only
  }
}
```
