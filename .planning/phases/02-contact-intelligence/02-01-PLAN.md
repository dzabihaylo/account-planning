function updateAccount(id, fields) {
  const allowedFields = ['name', 'sector', 'hq', 'revenue', 'employees', 'context', 'dot_color', 'display_revenue'];
  const setClauses = [];
  const values = [];
  for (const field of allowedFields) {
    if (fields[field] !== undefined) { setClauses.push(field + ' = ?'); values.push(fields[field]); }
  }
  setClauses.push("updated_at = datetime('now')");
  values.push(id);
  db.prepare('UPDATE accounts SET ' + setClauses.join(', ') + ' WHERE id = ?').run(...values);
  return db.prepare('SELECT * FROM accounts WHERE id = ?').get(id);
}
function deleteAccount(id) {
  const result = db.prepare("UPDATE accounts SET is_deleted = 1, updated_at = datetime('now') WHERE id = ? AND is_deleted = 0").run(id);
  if (result.changes === 0) return null;
  return { success: true };
}

| AI response -> database | AI-generated text stored and later rendered in browser |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | POST /api/accounts/:id/contacts | mitigate | Parameterized SQL in db.js createContact; influence CHECK constraint in schema |
| T-02-02 | Tampering | PUT /api/contacts/:id | mitigate | Parameterized SQL in db.js updateContact; allowedFields whitelist prevents updating arbitrary columns |
| T-02-03 | Tampering | POST /api/contacts/:id/outreach | mitigate | Parameterized SQL; channel/outcome CHECK constraints in schema; server-side validation of allowed values |
| T-02-04 | Elevation of Privilege | All new routes | mitigate | All routes placed after isAuthenticated() guard; cookie check on every request |
| T-02-05 | Information Disclosure | POST /api/contacts/:id/generate | accept | Account context sent to Anthropic API is intentional (AI generation requires it); API key stays server-side |
| T-02-06 | Tampering | AI response stored in DB | mitigate | AI text stored as-is in DB; XSS prevention happens at render time via escapeHtml (Plan 02 frontend responsibility) |
| T-02-07 | Denial of Service | POST /api/contacts/:id/generate | mitigate | Existing MAX_BODY (1MB) limit on request size; max_tokens: 2000 caps response size; rate limiting deferred (internal tool) |
