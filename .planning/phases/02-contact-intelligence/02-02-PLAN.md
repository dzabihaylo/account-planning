**3. Add staleness helper functions** (per D-12):
```javascript
function getStalenessClass(researchedAt) {
  if (!researchedAt) return 'stale-stale';
  var now = new Date();
  var researched = new Date(researchedAt);
  var days = Math.floor((now - researched) / (1000 * 60 * 60 * 24));
  if (days < 30) return 'stale-fresh';
  if (days < 90) return 'stale-aging';
  return 'stale-stale';
}

function getStalenessLabel(researchedAt) {
  if (!researchedAt) return 'STALE';
  var now = new Date();
  var researched = new Date(researchedAt);
  var days = Math.floor((now - researched) / (1000 * 60 * 60 * 24));
  if (days < 30) return 'FRESH';
  if (days < 90) return 'AGING';
  return 'STALE';
}
```

```javascript
function renderContactsGrid(accountId, contacts) {
  var panel = document.getElementById(accountId + '-contacts-panel');
  if (!panel) return;

  var html = '<div class="contacts-header">';
  html += '<span class="contacts-count">' + contacts.length + (contacts.length === 1 ? ' contact' : ' contacts') + '</span>';
  html += '<button class="btn btn-primary" onclick="openAddContactModal(\'' + escapeHtml(accountId) + '\')">Add Contact</button>';
  html += '</div>';

  if (contacts.length === 0) {
    html += '<div class="contacts-empty">';
    html += '<h3>No contacts yet</h3>';
    html += '<p>Add your first contact to start mapping decision-makers for this account.</p>';
    html += '<button class="btn btn-primary" onclick="openAddContactModal(\'' + escapeHtml(accountId) + '\')">Add Contact</button>';
    html += '</div>';
    panel.innerHTML = html;
    return;
  }

  var groups = { Champion: [], Evaluator: [], Blocker: [] };
  contacts.forEach(function(c) {
    if (groups[c.influence]) groups[c.influence].push(c);
  });

  var groupLabels = { Champion: 'Champions', Evaluator: 'Evaluators', Blocker: 'Blockers' };
  var groupOrder = ['Champion', 'Evaluator', 'Blocker'];

  groupOrder.forEach(function(influence) {
    if (groups[influence].length === 0) return;
    html += '<div class="contact-group">';
    html += '<div class="contact-group-hdr">' + groupLabels[influence] + '</div>';
    html += '<div class="contact-grid">';
    groups[influence].forEach(function(contact) {
      html += renderContactCard(contact);
    });
    html += '</div>';
    html += '</div>';
  });

  panel.innerHTML = html;
}
```

```javascript
var expandedContactId = null;

function toggleContactDetail(contactId, accountId) {
  // Collapse currently expanded card
  if (expandedContactId && expandedContactId !== contactId) {
    var prevDetail = document.getElementById('contact-detail-' + expandedContactId);
    var prevCard = document.getElementById('contact-card-' + expandedContactId);
    if (prevDetail) prevDetail.classList.remove('show');
    if (prevCard) prevCard.classList.remove('expanded');
  }

  var detail = document.getElementById('contact-detail-' + contactId);
  var card = document.getElementById('contact-card-' + contactId);
  if (!detail || !card) return;

  if (detail.classList.contains('show')) {
    detail.classList.remove('show');
    card.classList.remove('expanded');
    expandedContactId = null;
  } else {
    // Load detail content if empty
    if (!detail.innerHTML) {
      loadContactDetail(contactId, accountId);
    }
    detail.classList.add('show');
    card.classList.add('expanded');
    expandedContactId = contactId;
  }
}
```

```javascript
  if (!outreach || outreach.length === 0) return '<p style="color:var(--muted);font-size:13px">No outreach logged yet.</p>';

  outreach.forEach(function(entry) {

    html += '<div class="outreach-body">';
    html += '<div class="outreach-tags">';

    var outcomeColor = entry.outcome === 'connected' || entry.outcome === 'meeting scheduled' ? 'background:rgba(34,197,94,0.08);border:1px solid rgba(34,197,94,0.2);color:#86EFAC' : entry.outcome === 'no response' ? 'background:rgba(245,158,11,0.1);border:1px solid rgba(245,158,11,0.2);color:#FCD34D' : 'background:rgba(239,68,68,0.1);border:1px solid rgba(239,68,68,0.2);color:var(--red)';
    html += '<span class="outreach-tag" style="' + outcomeColor + '">' + escapeHtml(entry.outcome) + '</span>';

    html += '</div>';
    html += '</div>';
  });

  return html;
```
```javascript
}

function saveContact() {
  var name = document.getElementById('contactName').value.trim();
  var title = document.getElementById('contactTitle').value.trim();
  var influence = document.getElementById('contactInfluence').value;
  var valid = true;

  document.querySelectorAll('#contactModal .form-error').forEach(function(e) { e.style.display = 'none'; });

  if (!name) { document.getElementById('contactNameError').style.display = 'block'; valid = false; }
  if (!title) { document.getElementById('contactTitleError').style.display = 'block'; valid = false; }
  if (!influence) { document.getElementById('contactInfluenceError').style.display = 'block'; valid = false; }
  if (!valid) return;

  var editId = document.getElementById('contactEditId').value;
  var accountId = document.getElementById('contactAccountId').value;
  var data = {
    name: name, title: title, influence: influence,
    email: document.getElementById('contactEmail').value.trim(),
    linkedin: document.getElementById('contactLinkedin').value.trim(),
    phone: document.getElementById('contactPhone').value.trim()
  };

  var saveBtn = document.getElementById('contactModalSave');
  saveBtn.textContent = 'Saving...';
  saveBtn.disabled = true;

  var url = editId ? '/api/contacts/' + editId : '/api/accounts/' + encodeURIComponent(accountId) + '/contacts';
  var method = editId ? 'PUT' : 'POST';

  fetch(url, { method: method, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) })
    .then(function(r) {
      if (!r.ok) throw new Error('Save failed');
      return r.json();
    })
    .then(function() {
      closeContactModal();
      // Reload contacts grid
      var contactsPanel = document.getElementById(accountId + '-contacts-panel');
      if (contactsPanel) contactsPanel.dataset.loaded = '';
      loadContacts(accountId);
      // Clear expanded contact detail so it reloads fresh
      expandedContactId = null;
    })
    .catch(function() {
      saveBtn.textContent = editId ? 'Save Changes' : 'Save Contact';
      saveBtn.disabled = false;
    });
}
```
