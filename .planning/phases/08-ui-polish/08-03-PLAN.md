Add a `renameCategory(oldName, newName)` function using `db.transaction()` for atomicity:

```javascript
function renameCategory(oldName, newName) {
  var rename = db.transaction(function() {
    db.prepare(
      "UPDATE accounts SET sector = ?, updated_at = datetime('now') WHERE sector = ? AND is_deleted = 0"
    ).run(newName, oldName);
  });
  rename();
  return db.prepare(
    'SELECT COUNT(*) as count FROM accounts WHERE sector = ? AND is_deleted = 0'
  ).get(newName);
}
```

Export `renameCategory` in `module.exports`.

  var existing = document.querySelector('.gd-toast');
  if (existing) existing.remove();
  var toast = document.createElement('div');
  toast.className = 'gd-toast gd-toast-' + (variant || 'error');

**B. Add startCategoryRename() and confirmCategoryRename() functions:**

```javascript
function startCategoryRename(span) {
  var oldName = span.textContent;
  var input = document.createElement('input');
  input.type = 'text';
  input.value = oldName;
  input.className = 'sidebar-rename-input';
  span.replaceWith(input);
  input.focus();
  input.select();

  var committed = false;
  function commit() {
    if (committed) return;
    committed = true;
    var newName = input.value.trim();
    if (!newName || newName === oldName) {
      // No change or empty — cancel
      input.replaceWith(span);
      return;
    }
    confirmCategoryRename(oldName, newName);
  }
  function cancelRename() {
    if (committed) return;
    committed = true;
    input.replaceWith(span);
  }

  input.addEventListener('keydown', function(e) {
    if (e.key === 'Enter') { e.preventDefault(); commit(); }
    if (e.key === 'Escape') { cancelRename(); }
  });
  input.addEventListener('blur', commit);
}

async function confirmCategoryRename(oldName, newName) {
  // Update sectionState before re-render (per Pitfall 2 from RESEARCH.md)
  if (sectionState.hasOwnProperty(oldName)) {
    sectionState[newName] = sectionState[oldName];
    delete sectionState[oldName];
  }

  try {
    var resp = await fetch('/api/categories/rename', {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ oldName: oldName, newName: newName })
    });
    if (!resp.ok) throw new Error('rename failed');

    // Clear rendered panels so they re-render with updated sector name (per Pitfall 3)
    document.getElementById('mainContent').innerHTML = '';
    await loadAccounts();
    if (currentAccount) showAccount(currentAccount);
  } catch (e) {
    // Revert sectionState on failure
    if (sectionState.hasOwnProperty(newName)) {
      sectionState[oldName] = sectionState[newName];
      delete sectionState[newName];
    }
    showToast('Rename failed. Please try again.', 'error');
    await loadAccounts();
  }
}
```

CRITICAL implementation notes:
- The `committed` boolean prevents the double-fire of blur+Enter (Pitfall 1 from RESEARCH.md)
- `sectionState` is updated before `loadAccounts()` so the collapse state follows the renamed category (Pitfall 2)
- `mainContent.innerHTML = ''` forces panel re-render to show new sector name (Pitfall 3)
