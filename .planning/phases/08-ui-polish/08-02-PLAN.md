**B. Add populateCategoryDropdown() function (in the JS section, near openEditModal):**

```javascript
async function populateCategoryDropdown(currentSector) {
  var select = document.getElementById('modalSector');
  var sectors = [];
  try {
    var resp = await fetch('/api/categories');
    sectors = await resp.json();
  } catch (e) { /* fall back to empty list */ }

  var options = '<option value="">Select a category</option>';
  for (var i = 0; i < sectors.length; i++) {
    var s = sectors[i];
    options += '<option value="' + escapeHtml(s) + '"' + (s === currentSector ? ' selected' : '') + '>' + escapeHtml(s) + '</option>';
  }
  options += '<option value="__new__">New category...</option>';
  select.innerHTML = options;

  // If currentSector exists but is not in the list, add it
  if (currentSector && sectors.indexOf(currentSector) === -1) {
    var opt = document.createElement('option');
    opt.value = currentSector;
    opt.textContent = currentSector;
    opt.selected = true;
    select.insertBefore(opt, select.lastElementChild);
  }

  // Wire the "New category..." toggle
  var newInput = document.getElementById('modalSectorNew');
  newInput.style.display = 'none';
  newInput.value = '';
  select.onchange = function() {
    if (select.value === '__new__') {
      newInput.style.display = '';
      newInput.focus();
    } else {
      newInput.style.display = 'none';
      newInput.value = '';
    }
  };
}
```
