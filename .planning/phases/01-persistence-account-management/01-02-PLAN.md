New JS structure:

```javascript
let accounts = [];        // populated from API
let chatHistories = {};
let currentAccount = null;
let editingAccountId = null;
let removingAccountId = null;

// ---- Data Loading ----
async function loadAccounts() {
  const res = await fetch('/api/accounts');
  accounts = await res.json();
  renderSidebar();
  if (accounts.length > 0) {
    showAccount(accounts[0].id);
  } else {
    renderEmptyState();
  }
  document.getElementById('acctCount').textContent = accounts.length + ' account' + (accounts.length !== 1 ? 's' : '');
}

// ---- Sidebar Rendering ----
function renderSidebar() {
  const sidebar = document.getElementById('sidebar');
  // Group accounts by sector
  const groups = {};
  accounts.forEach(a => {
    if (!groups[a.sector]) groups[a.sector] = [];
    groups[a.sector].push(a);
  });
  let html = '';
  for (const [sector, items] of Object.entries(groups)) {
    html += '<div class="sidebar-section">' + sector + '</div>';
    items.forEach(a => {
      html += '<button class="sidebar-item" onclick="showAccount(\'' + a.id + '\',this)" id="sb-' + a.id + '">';
      html += '<span class="sidebar-dot" style="background:' + a.dot_color + '"></span>';
      html += a.name.split(' ')[0];  // Short name (first word or company name)
      // Use display_revenue for sidebar
      html += '<span class="sidebar-revenue">' + a.display_revenue + '</span>';
      html += '</button>';
    });
  }
  html += '<button class="sidebar-add" onclick="openAddModal()">+ Add Account</button>';
  sidebar.innerHTML = html;
}
```
