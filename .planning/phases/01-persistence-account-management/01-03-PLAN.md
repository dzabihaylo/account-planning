**Route: POST /api/accounts/:id/chat**
Saves a single chat message. Expects body: `{ role: "user"|"assistant", content: "..." }`.
```javascript
// Match: POST /api/accounts/{id}/chat
const chatPostMatch = parsed.pathname.match(/^\/api\/accounts\/([a-z0-9-]+)\/chat$/);
if (req.method === 'POST' && chatPostMatch) {
  let body = '';
  req.on('data', chunk => { body += chunk; });
  req.on('end', () => {
    let parsed_body;
    try {
      parsed_body = JSON.parse(body);
    } catch (e) {
      res.writeHead(400, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ error: 'Invalid JSON' }));
      return;
    }
    const { role, content } = parsed_body;
    if (!role || !content || !['user', 'assistant'].includes(role)) {
      res.writeHead(400, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ error: 'role (user|assistant) and content are required' }));
      return;
    }
    const msg = db.addChatMessage(chatPostMatch[1], role, content);
    res.writeHead(201, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(msg));
  });
  return;
}
```

**Route: DELETE /api/accounts/:id/chat**
Clears all chat history for an account (optional utility).
```javascript
const chatDeleteMatch = parsed.pathname.match(/^\/api\/accounts\/([a-z0-9-]+)\/chat$/);

  db.clearChatMessages(chatDeleteMatch[1]);
  res.writeHead(200, { 'Content-Type': 'application/json' });
  res.end(JSON.stringify({ success: true }));
  return;
}
```

**Add `loadChatHistory(id)` function** that fetches and renders saved messages:
```javascript
async function loadChatHistory(id) {
  try {
    const res = await fetch('/api/accounts/' + id + '/chat');
    const messages = await res.json();
    if (messages.length === 0) return;
    // Populate in-memory chatHistories from DB
    chatHistories[id] = messages.map(m => ({ role: m.role, content: m.content }));
    // Render each message into the DOM
    const msgsContainer = document.getElementById('msgs-' + id);
    if (!msgsContainer) return;
    messages.forEach(m => {
      addMsg(id, m.role === 'user' ? 'user' : 'ai', m.content);
    });
    // Hide chips if there's existing history
    const chips = document.getElementById('chips-' + id);
    if (chips) chips.style.display = 'none';
  } catch (e) {
    console.error('Failed to load chat history:', e);
  }
}
```

**Add `saveChatMessage(accountId, role, content)` function** that persists to DB:
```javascript
function saveChatMessage(accountId, role, content) {
  fetch('/api/accounts/' + accountId + '/chat', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ role: role, content: content })
  }).catch(e => console.error('Failed to save chat message:', e));
}
```
This is fire-and-forget — don't await it, don't block the UI. Chat still works if save fails; it just won't persist.
